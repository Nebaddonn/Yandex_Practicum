def diff(positive1, positive2, test1, test2):
    #     :
    p1 = positive1/test1
    #     :
    p2 = positive2/test2
    #     :
    p_combined = (positive1 + positive2) / (test1 + test2)
    #    
    difference = p1 - p2 
    #    .   
    z_value = difference / (p_combined * (1 - p_combined) * (1/test1 + 1/test2)) ** 0.5
    #     ( 0, . 1)
    distr = st.norm(0, 1)  
    p_value = (1 - distr.cdf(abs(z_value))) * 2
    return p_value
